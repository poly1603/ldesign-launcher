## 🚀 使用指南

### 快速开始
```bash
# 安装
npm install -g @ldesign/launcher

# 初始化项目
launcher create my-app --template vue-spa

# 启动开发
cd my-app
launcher dev --incremental --dashboard

# 安全扫描
launcher scan --all

# 构建发布
launcher build --optimize --analyze
```

### 高级配置示例
```typescript
// launcher.config.ts
import { defineConfig } from '@ldesign/launcher'

export default defineConfig({
  // 增量构建
  incrementalBuild: {
    enabled: true,
    concurrency: 4,
    cacheStrategy: 'content-hash'
  },
  
  // 分布式缓存
  distributedCache: {
    enabled: true,
    redis: {
      cluster: ['redis://node1:6379', 'redis://node2:6379'],
      namespace: 'team-frontend'
    }
  },
  
  // 实时监控
  monitoring: {
    dashboard: {
      enabled: true,
      port: 3001,
      auth: true
    },
    alerts: {
      channels: ['email', 'dingtalk'],
      rules: [
        {
          metric: 'buildTime',
          threshold: 60000,
          severity: 'warning'
        }
      ]
    }
  },
  
  // 安全配置
  security: {
    scan: {
      onCommit: true,
      breakOnCritical: true
    },
    allowedLicenses: ['MIT', 'Apache-2.0'],
    customRules: [
      {
        pattern: 'eval\\(',
        severity: 'critical',
        message: '禁止使用 eval'
      }
    ]
  },
  
  // 插件配置
  plugins: [
    '@ldesign/plugin-vue-optimizer',
    '@ldesign/plugin-bundle-analyzer',
    {
      name: 'custom-plugin',
      options: { /* ... */ }
    }
  ]
})
```
