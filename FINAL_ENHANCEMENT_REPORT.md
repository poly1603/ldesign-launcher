## 🚀 使用示例

### 完整配置
```typescript
import { ViteLauncher } from '@ldesign/launcher'

const launcher = new ViteLauncher({
  // 基础配置
  root: './src',
  
  // 增量构建
  incrementalBuild: {
    enabled: true,
    concurrency: 4,
    analyzeDependencies: true
  },
  
  // 分布式缓存
  distributedCache: {
    enabled: true,
    adapter: 'redis',
    namespace: 'team-alpha',
    syncStrategy: 'immediate'
  },
  
  // 监控仪表板
  dashboard: {
    enabled: true,
    port: 3001,
    updateInterval: 1000
  },
  
  // 告警系统
  alerting: {
    enabled: true,
    evaluationInterval: 30,
    channels: ['email', 'dingtalk']
  },
  
  // 配置编辑器
  configEditor: {
    enabled: true,
    port: 3002,
    autoSaveInterval: 30
  }
})

await launcher.initialize()
```

### CLI 命令
```bash
# 使用增量构建
launcher build --incremental

# 启动监控仪表板
launcher monitor --dashboard

# 打开配置编辑器
launcher config --editor

# 从模板创建项目
launcher create my-app --template vue-spa

# 测试告警通道
launcher alert test --channel email
```
